    <script>
        let currentMethod = 'auto';
        
        function selectMethod(method) {
            currentMethod = method;
            document.querySelectorAll('.method-card').forEach(card => {
                card.classList.remove('active');
            });
            event.target.closest('.method-card').classList.add('active');
            
            document.querySelectorAll('.step').forEach(step => {
                step.classList.remove('active');
            });
            document.getElementById(method).classList.add('active');
            
            updateProgressBar();
        }
        
        function nextStep() {
            if (currentMethod === 'auto' || currentMethod === 'copy' || currentMethod === 'manual') {
                document.querySelectorAll('.step').forEach(step => {
                    step.classList.remove('active');
                });
                document.getElementById('success').classList.add('active');
                currentMethod = 'success';
                document.getElementById('nextBtn').style.display = 'none';
                document.getElementById('prevBtn').style.display = 'block';
                updateProgressBar();
            }
        }
        
        function previousStep() {
            if (currentMethod === 'success') {
                document.querySelectorAll('.step').forEach(step => {
                    step.classList.remove('active');
                });
                document.getElementById('auto').classList.add('active');
                currentMethod = 'auto';
                document.getElementById('nextBtn').style.display = 'block';
                document.getElementById('prevBtn').style.display = 'none';
                updateProgressBar();
            }
        }
        
        function updateProgressBar() {
            const methods = ['auto', 'copy', 'manual', 'success'];
            const index = methods.indexOf(currentMethod);
            const progress = ((index + 1) / methods.length) * 100;
            document.getElementById('progressBar').style.width = progress + '%';
        }
        
        function copyToClipboard(text, button) {
            navigator.clipboard.writeText(text).then(() => {
                const originalText = button.textContent;
                button.textContent = '‚úì Copied!';
                button.classList.add('copied');
                setTimeout(() => {
                    button.textContent = originalText;
                    button.classList.remove('copied');
                }, 2000);
            });
        }
        
        updateProgressBar();
    </script>
